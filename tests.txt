### link-distance

distance-to-self
  extensions [network]
  O> crt 1
  [network:link-distance turtle 0 links] of turtle 0 => 0

distance-two-turtles-no-links
  extensions [network]
  O> crt 2
  [network:link-distance turtle 0 links] of turtle 1 => -1
  [network:link-distance turtle 1 links] of turtle 0 => -1

distance-one-undirected-link
  extensions [network]
  O> crt 2 [ create-links-with other turtles ]
  [network:link-distance turtle 0 links] of turtle 1 => 1
  [network:link-distance turtle 1 links] of turtle 0 => 1

distance-one-directed-link
  extensions [network]
  O> crt 2
  O> ask turtle 0 [ create-link-to turtle 1 ]
  [network:link-distance turtle 0 links] of turtle 1 => -1
  [network:link-distance turtle 1 links] of turtle 0 => 1

distance-always-one-in-fully-connected-network
  extensions [network]
  to-report distances report [network:link-distance myself links] of link-neighbors end
  O> crt 10 [ create-links-with other turtles ]
  remove-duplicates reduce sentence [distances] of turtles => [1]

distance-along-chain
  extensions [network]
  O> crt 5 [ if who < 4 [ create-link-with turtle (who + 1) ] ]
  sort [network:link-distance turtle 0 links] of turtles => [0 1 2 3 4]
  sort [network:link-distance turtle 4 links] of turtles => [0 1 2 3 4]
  sort remove-duplicates [network:link-distance turtle 2 links] of turtles => [0 1 2]

distance-around-circle
  extensions [network]
  O> crt 8 [ create-link-with turtle ((who + 1) mod 8) ]
  sort [network:link-distance turtle 0 links] of turtles => [0 1 1 2 2 3 3 4]

distance-around-two-circles
  extensions [network]
  O> crt 8 [ create-link-with turtle ((who + 1) mod 8) ]
  O> crt 8 [ create-link-with turtle ((who - 8 + 1) mod 8 + 8) ]
  sort [network:link-distance turtle 0 links] of turtles with [who < 8] => [0 1 1 2 2 3 3 4]
  sort [network:link-distance turtle 8 links] of turtles with [who >= 8] => [0 1 1 2 2 3 3 4]
  remove-duplicates [network:link-distance turtle 8 links] of turtles with [who < 8] => [-1]
  remove-duplicates [network:link-distance turtle 0 links] of turtles with [who >= 8] => [-1]

### extended-link-neighbors

neighbordhood-negative-radius
  extensions [network]
  O> crt 1
  [network:extended-link-neighbors -1 links] of turtle 0 => ERROR Extension exception: radius cannot be negative

neighborhood-zero-radius
  extensions [network]
  O> crt 5
  remove-duplicates [count network:extended-link-neighbors 0 links] of turtles => [1]

neighborhood-two-circles

  O> crt 8 [ create-link-with turtle ((who + 1) mod 8) ]
  O> crt 8 [ create-link-with turtle ((who - 8 + 1) mod 8 + 8) ]
  remove-duplicates [count network:extended-link-neighbors 0 links] of turtles => [1]
  remove-duplicates [count network:extended-link-neighbors 1 links] of turtles => [3]
  remove-duplicates [count network:extended-link-neighbors 2 links] of turtles => [5]
  remove-duplicates [count network:extended-link-neighbors 3 links] of turtles => [7]
  remove-duplicates [count network:extended-link-neighbors 4 links] of turtles => [8]
  remove-duplicates [count network:extended-link-neighbors 5 links] of turtles => [8]
  remove-duplicates [count network:extended-link-neighbors 1E38 links] of turtles => [8]

### path-turtles

path-turtles-self
  extensions [network]
  O> crt 1
  map [[who] of ?] [network:path-turtles turtle 0 links] of turtle 0 => [0]

path-turtles-no-path
  extensions [network]
  O> crt 2
  map [[who] of ?] [network:path-turtles turtle 1 links] of turtle 0 => []
  map [[who] of ?] [network:path-turtles turtle 0 links] of turtle 1 => []

path-turtles-undirected-pair
  extensions [network]
  O> crt 2 [ create-links-with other turtles ]
  map [[who] of ?] [network:path-turtles turtle 1 links] of turtle 0 => [0 1]
  map [[who] of ?] [network:path-turtles turtle 0 links] of turtle 1 => [1 0]

path-turtles-fully-connected
  extensions [network]
  O> crt 6 [ create-links-with other turtles ]
  sort [length network:path-turtles turtle 0 links] of turtles => [1 2 2 2 2 2]

path-turtles-on-a-circle
  extensions [network]
  O> crt 6 [ create-link-with turtle ((who + 1) mod 6) ]
  sort [length network:path-turtles turtle 0 links] of turtles => [1 2 2 3 3 4]
